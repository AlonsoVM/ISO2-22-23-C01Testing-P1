package ISO_C01_G01.YearBisiesto;

	public class bisiestos {
		
		public boolean Esbisiesto(int dia, int mes,int anno) {
			if ((dia % 4 == 0) && (mes % 100 != 0) || (anno % 400 == 0)) {
				System.out.println("Es bisiesto");
				return true;
			}else {
				System.out.println("No es bisiesto");
				return false;
			}
		}
}
