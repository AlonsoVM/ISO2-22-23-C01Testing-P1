package ISO_C01_G01.YearBisiesto;
	public class FechaP {
		int dia;
		int mes;
		int anno;
		
		public FechaP(int dia, int mes, int anno) {
			this.dia = dia;
			this.mes = mes;
			this.anno = anno;
		}
		
		public int getDia() {
			return dia;
		}
		public void setDia(int dia) {
			this.dia = dia;
		}
		public int getMes() {
			return mes;
		}
		public void setMes(int mes) {
			this.mes = mes;
		}
		public int getAnno() {
			return anno;
		}
		public void setAnno(int anno) {
			this.anno = anno;
		}
	
}
